有17个人围成一圈(编号为0~16)，从第0号的人开始从1报数，凡报到3的倍数的人离开圈子，然后再数下去，直到最后只剩下一个人为止，问此人原来的位置是多少号？
(字符实现) 此方法受字符限制如300人
Private Sub Form_click()
   Dim s As String
   Dim y As String
   Dim i As Integer
   s = "ABCDEFGHIJKLMNOPQ"
   y = s
   Do Until Len(s) = 1
      i = i + 1
      If i Mod 3 = 0 Then s = Replace(s, Mid(y, i, 1), "")
      If i = Len(y) Then y = y & s
   Loop
   Print InStr(y, s) - 1
End Sub

返回常用三角函数值
'1.弧度转角度
'2.正负角转正周角
'3.转锐角
'4.查表返回值
's 三角函数名, n 角度或弧度, b 是否是弧度
Function v(s, n, Optional b = False)
   If b Then n = n * 180 / 3.14
   Do While n > 360
      n = n - 360
   Loop
   Do While n < 0
      n = n + 360
   Loop
   Select Case n
      Case 0 To 90
         v = z(s, n)
      Case 90 To 180
         n = 180 - n
         If s = "sin" Then v = z(s, n)
         If s = "cos" Then v = z(s, n, "-")
         If s = "tan" Then v = z(s, n, "-")
      Case 180 To 270
         n = n - 180
         If s = "sin" Then v = z(s, n, "-")
         If s = "cos" Then v = z(s, n, "-")
         If s = "tan" Then v = z(s, n)
      Case 270 To 360
         n = Int(360 - n)
         If s = "sin" Then v = z(s, n, "-")
         If s = "cos" Then v = z(s, n)
         If s = "tan" Then v = z(s, n, "-")
   End Select
End Function
's 三角函数名, n 角度, f 正负符号
Function z(s, n, Optional f = "")
   Select Case Int(n)
      Case 0
         If s = "sin" Then z = f + "0"
         If s = "cos" Then z = f + "1"
         If s = "tan" Then z = f + "0"
      Case 30
         If s = "sin" Then z = f + "1/2"
         If s = "cos" Then z = f + "√3/2"
         If s = "tan" Then z = f + "√3/3"
      Case 45
         If s = "sin" Then z = f + "√2/2"
         If s = "cos" Then z = f + "√2/2"
         If s = "tan" Then z = f + "1"
      Case 60
         If s = "sin" Then z = f + "√3/2"
         If s = "cos" Then z = f + "1/2"
         If s = "tan" Then z = f + "√3"
      Case 90
         If s = "sin" Then z = f + "1"
         If s = "cos" Then z = f + "0"
         If s = "tan" Then z = "无意义"
      Case Else
         z = "此度数不常用！！！"
   End Select
End Function

Python 快速排序
def quick_sort(array):
    if len(array) < 2:
        return array
    else:
        pivot = array[0]
        less_than_pivot = [x for x in array if x <= pivot]
        more_than_pivot = [x for x in array if x > pivot]
        return quick_sort(less_than_pivot) + [pivot] + quick_sort(more_than_pivot)
